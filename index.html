import SwiftUI struct ContentView: View { // Μεταβλητή για το πόσο νερό έχει πιει ο χρήστης @State private var currentIntake: Int = 0 // Το όριο των 3 λίτρων σε χιλιοστόλιτρα (ml) let maxCapacity: Int = 3000 // Υπολογισμός του ποσοστού για το γέμισμα του μπουκαλιού var progress: CGFloat { return CGFloat(currentIntake) / CGFloat(maxCapacity) } var body: some View { VStack(spacing: 30) { // Τίτλος Text("Water Tracker") .font(.largeTitle) .bold() .foregroundColor(.blue) .padding(.top, 40) // Το Ψηφιακό Μπουκάλι ZStack(alignment: .bottom) { // Το background του μπουκαλιού (άδειο) RoundedRectangle(cornerRadius: 30) .fill(Color.blue.opacity(0.1)) .frame(width: 160, height: 350) .overlay( RoundedRectangle(cornerRadius: 30) .stroke(Color.blue, lineWidth: 4) ) // Το νερό που γεμίζει το μπουκάλι RoundedRectangle(cornerRadius: 26) .fill(Color.blue.opacity(0.6)) .frame(width: 152, height: 342 * min(progress, 1.0)) .padding(4) .animation(.easeInOut(duration: 0.4), value: currentIntake) } // Ενδείξεις Ποσότητας VStack(spacing: 5) { Text("\(currentIntake) / \(maxCapacity) ml") .font(.title2) .bold() Text(String(format: "%.2f / 3.0 L", Double(currentIntake) / 1000.0)) .font(.subheadline) .foregroundColor(.secondary) } // Κουμπιά Προσθήκης Νερού HStack(spacing: 20) { Button(action: { addWater(amount: 250) }) { Text("+250 ml") .font(.headline) .foregroundColor(.white) .padding() .frame(maxWidth: .infinity) .background(Color.blue) .cornerRadius(15) } Button(action: { addWater(amount: 500) }) { Text("+500 ml") .font(.headline) .foregroundColor(.white) .padding() .frame(maxWidth: .infinity) .background(Color.blue) .cornerRadius(15) } } .padding(.horizontal, 30) Spacer() // Κουμπί Reset Button(action: { currentIntake = 0 }) { Text("Επαναφορά") .font(.body) .bold() .foregroundColor(.red) .padding() } .padding(.bottom, 20) } } // Συνάρτηση για την προσθήκη νερού χωρίς να ξεπερνάει το όριο func addWater(amount: Int) { if currentIntake + amount <= maxCapacity { currentIntake += amount } else { currentIntake = maxCapacity // Σταματάει στα 3000ml } } } // Προεπισκόπηση για το Xcode struct ContentView_Previews: PreviewProvider { static var previews: some View { ContentView() } }
